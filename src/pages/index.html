McFly is a full-stack no-framework framework that assists developers in leveraging the web platform.

Here's a sample interactive custom element: 

class HelloWorld extends HTMLElement {
  static observedAttributes = ["name"];

  connectedCallback() {
    let count = 0;
    this.onclick = () => {
      this.setAttribute("name", `Clicked ${++count}x`);
    };
    this.setAttribute("title", "Click me please");
  }

  attributeChangedCallback(property, previousValue, currentValue) {
    if (previousValue !== currentValue) {
      this.innerHTML = ``;
    }
  }
}
